// Common Revenant
ACTOR CommonRevenantRPG : CommonRevenant Replaces CommonRevenant
{
	Tag "Revenant"

	States
	{
	Spawn:
		SKEL A 0 NoDelay ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::Spawn
	See:
		SKEL A 0 ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::See
	}
}
